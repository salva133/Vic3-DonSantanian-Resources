### Religion heritage triggers
# Example: ROOT = { is_christian = yes }
is_christian = {
	religion = {
		has_discrimination_trait = heritage_christian
	}
}

# Example: any_scope_pop = { is_jewish = yes }
is_jewish = {
	religion = {
		has_discrimination_trait = heritage_jewish
	}
}

# Example: ROOT = { is_muslim = yes }
is_muslim = {
	religion = {
		has_discrimination_trait = heritage_islamic
	}
}

# Example: ROOT = { is_abrahamic_non_mohammedan = yes }
is_abrahamic_non_mohammedan = {
	OR = {
		is_christian = yes
		is_jewish = yes
	}
}

### Religion-nation triggers
# Example: any_country = { limit = { country_is_jewish_nation = yes } }
country_is_jewish_nation = {
	custom_description = {
		text = country_is_jewish_nation_desc
		subject = this
		any_primary_culture = {
			count = all
			has_discrimination_trait = heritage_jewish
		}
		country_has_state_religion = rel:jewish
	}
}

# Example: ROOT = { country_is_a_judeo_christian_nation = yes }
country_is_a_judeo_christian_nation = {
	custom_description = {
		text = country_is_a_judeo_christian_nation_desc
		subject = this
		religion = {
			OR = {
				has_discrimination_trait = heritage_christian
				has_discrimination_trait = heritage_jewish
			}
		}
	}
}

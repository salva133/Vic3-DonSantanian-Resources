# Detects a fascist government by its legal and heraldic silhouette. Two shapes
# count: (1) a republican facade (presidential / parliamentary / corporate) with
# an autocratic executive and a fascist coat of arms, or (2) a corporatist
# autocracy that is not a Germanic ethnostate, not a non-ethnostate Austria, and
# not a Hispanophone-cultured country (those read as other strains of fascism).
country_has_fascist_government_trigger = {
	custom_description = {
		text = country_has_fascist_government_trigger_desc
		subject = this
		OR = {
			AND = {
				OR = {
					has_law_or_variant = law_type:law_presidential_republic
					has_law_or_variant = law_type:law_parliamentary_republic
					has_law_or_variant = law_type:law_corporate_state
				}
				OR = {
					has_law_or_variant = law_type:law_autocracy
					has_law_or_variant = law_type:law_technocracy
					has_law_or_variant = law_type:law_oligarchy
					has_law_or_variant = law_type:law_single_party_state
				}
				coa_fascist_trigger = yes
			}
			AND = {
				has_law_or_variant = law_type:law_corporate_state
				OR = {
					has_law_or_variant = law_type:law_autocracy
					has_law_or_variant = law_type:law_oligarchy
					has_law_or_variant = law_type:law_single_party_state
				}
				NOR = {
					AND = {
						has_law_or_variant = law_type:law_ethnostate
						OR = {
							country_has_primary_culture = cu:north_german
							country_has_primary_culture = cu:south_german
						}
					}
					AND = {
						NOT = {
							has_law_or_variant = law_type:law_ethnostate
						}
						c:AUS ?= this
					}
					any_primary_culture = {
						has_discrimination_trait = language_hispanophone
					}
				}
			}
		}
	}
}

is_welfare_state_with_low_legitimacy = {
	custom_description = {
		text = is_welfare_state_with_low_legitimacy_desc
		subject = this
		has_law_or_variant = law_type:law_old_age_pension
		has_law_or_variant = law_type:law_worker_protections
		institution_investment_level = {
			institution = institution_social_security
			value = 5
		}
		institution_investment_level = {
			institution = institution_workplace_safety
			value = 5
		}
		government_legitimacy < 25
	}
}

	custom_description = {
		text = country_eligible_as_mass_migration_source_country_desc
		subject = this
		country_can_have_mass_migration_to = root
		any_scope_culture = {
			OR = {
				shares_heritage_trait_with_any_primary_culture = root
				shares_language_trait_with_any_primary_culture = root
			}
			any_scope_pop = {
				count > 1000
			}
		}
	}
}

### State control triggers
# Example: any_scope_state = { state_is_occupied = yes }
state_is_occupied = {
	custom_description = {
		text = state_is_occupied_desc
		subject = this
		NOT = { controller = owner }
	}
}

# Example: any_scope_state = { state_is_occupied_by_root = yes }
state_is_occupied_by_root = {
	custom_description = {
		text = state_is_occupied_by_root_desc
		subject = this
		controller = root
	}
}

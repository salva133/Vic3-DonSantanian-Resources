### State-scoped helpers

# Macro helper: tag a state region with a state trait when none of its
# contained states already carries it. Idempotent across re-runs so this is
# safe to call from on_game_started even when traits were applied previously.
# Callable from any scope.
# Example: add_state_trait_to_region_if_missing = { STATE = STATE_NORMANDY TRAIT = state_trait_natural_harbors }
add_state_trait_to_region_if_missing = {
	s:$STATE$ = {
		if = {
			limit = {
				any_scope_state = {
					NOT = { has_state_trait = $TRAIT$ }
				}
			}
			add_state_trait = $TRAIT$
		}
	}
}

### Pop consolidation

# Moves every pop of religion $RELIGION$ from every state in the world into the
# state in ROOT scope. Call from a state scope; ROOT is the destination state.
# Wrap the call in a custom_tooltip at the call site to replace the generated
# per-pop tooltip with a single line.
# ROOT     = destination state
# RELIGION = religion key without the rel: prefix, e.g. jewish
# Example: capital = { state_gather_all_pops_of_religion = { RELIGION = jewish } }
state_gather_all_pops_of_religion = {
	every_state = {
		limit = {
			any_scope_pop = {
				has_pop_religion = $RELIGION$
			}
		}
		every_scope_pop = {
			limit = {
				has_pop_religion = $RELIGION$
			}
			move_pop = root
		}
	}
}

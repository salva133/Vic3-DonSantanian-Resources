### World / country surgery

# Turns every country beyond the top 200 by prestige into a decentralized
# country. Callable from any scope; leaves the country_rank global variable
# behind.
decentralize_countries = {
	set_global_variable = {
		name = country_rank
		value = 0
	}
	ordered_country = {
		limit = {
			NOT = { is_country_type = decentralized }
		}
		order_by = prestige
		check_range_bounds = no
		change_global_variable = {
			name = country_rank
			add = 1
		}
		if = {
			limit = {
				global_var:country_rank >= 200
			}
			set_country_type = decentralized
		}
	}
}

# Exiles the least popular agitator of the country in scope, saving them as
# scope:exiled_character for follow-up script.
exile_least_popular_agitator = {
	if = {
		limit = {
			any_scope_character = {
				has_role_of_type = agitator
			}
		}
		ordered_scope_character = {
			check_range_bounds = no
			limit = { has_role_of_type = agitator }
			order_by = popularity
			position = 99
			save_scope_as = exiled_character
			exile_character = yes
		}
	}
}

### Pop surgery (country scope)

# Kills 99% of every non-primary-culture pop in every state of the country in
# scope, then starts enacting Slavery Banned if slavery is still legal.
country_kill_all_non_primary_pops_in_all_states = {
	every_scope_state = {
		state_kill_all_non_primary_pops = yes
	}
	if = {
		limit = {
			NOT = {
				has_law_or_variant = law_type:law_slavery_banned
			}
		}
		start_enactment = law_type:law_slavery_banned
		add_enactment_phase = 3
	}
}

# Caps pop literacy at 20% across the country in scope, with academics and
# engineers retained at 60%. Models a literacy collapse.
collapse_pop_literacy = {
	every_scope_pop = {
		limit = { literacy_rate > 0.2 }
		set_pop_literacy = {
			literacy_rate = {
				value = 0.2
				if = {
					limit = {
						OR = {
							is_pop_type = academics
							is_pop_type = engineers
						}
					}
					add = 0.4
				}
			}
		}
	}
}
